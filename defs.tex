\newcommand{\cmdquote}[1]{% команда для рисования кнопок клавиатуры
	\begin{minipage}{0.9\textwidth} 
		\fbox{
			\parbox{\textwidth}{    
				\texttt{\$ #1}
			}
		}
	\end{minipage}  
}   

\newcommand*\keystroke[1]{% команда для рисования кнопок клавиатуры
	\tikz[baseline=(key.base)]
	\node[%
	draw,
	fill=white,
	drop shadow={shadow xshift=0.25ex,shadow yshift=-0.25ex,fill=black,opacity=0.75},
	rectangle,
	rounded corners=2pt,
	inner sep=1pt,
	line width=0.5pt,
	font=\scriptsize\sffamily
	](key) {#1\strut}
	;
}
